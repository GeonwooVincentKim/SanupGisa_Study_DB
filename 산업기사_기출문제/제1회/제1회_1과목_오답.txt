void func(char* str) {
  int i = 1, cnt = 0;
  while (cnt++ < 5) {
    if (i > 7) i = i - 4;
    printf("%c", *(str + i));
    i = i + 3;
  }
}

main () {
  char str[] = "ABCDEFGH";
  func(str);
}

- 1. str[] = "ABCDEFGH";
- 2. cnt = 0, cnt = 1 < 5 (true)
- 3. i = 1, *(str + 1) -> B
- 4. i = 1 + 3 -> 4

- 5. cnt = 1, cnt = 2 < 5 (true)
- 6. i = 4, *(str + 4) -> E
- 7. i = 4 + 3 -> 7

- 8. cnt = 2, cnt = 3 < 5 (true)
- 9. i = 7, *(str + 7) -> H
- 10. i = 7 + 3 -> 10

- 11. cnt = 3, cnt = 4 < 5 (true)
- 12. i = 10, (i > 7) i = 10 - 4 -> 6;
- 13. i = 6, *(str + 6) -> G
- 14. i = 6 + 3 -> 9

- 14. cnt = 4, cnt = 5 < 5 (true)
- 15. i = 9, (i > 7) i = 9 - 4 -> 5;
- 16. i = 5, *(str + 5) -> F
- 17. i = 5 + 3 -> 8

- 18. cnt = 5, cnt = 6 (false)

-> BEHGF

void func(int *a) {
  int x = 1;
  for (int i = 0; i >= 0; i += x) {
    *(a + i) += i;
   
    if (i >= 4) x = -1;
  }
}

main() {
  int arry[] = {1, 2, 3, 2, 1};
  func(arry);

  for (int i = 4; i >= 0; i--) printf("%d ", arry[i]);
}

- 1. arry = {1, 2, 3, 2, 1};
- 2. x = 1; for (int i = 0; i >= 0; i += x) i = 0
